91 template <class T, class U>
92 SPECTRE_ALWAYS_INLINE T narrow(U u) {
93  T t = narrow_cast<T>(u);
94  if (static_cast<U>(t) != u) {
95  ERROR("Failed to cast " << u << " of type " << pretty_type::get_name<U>()
96  << " to type " << pretty_type::get_name<T>());
97  }
98  if (not details::is_same_signedness<T, U>::value and
99  ((t < T{}) != (u < U{}))) {
100  ERROR("Failed to cast " << u << " of type " << pretty_type::get_name<U>()
101  << " to type " << pretty_type::get_name<T>());
102  }
103  return t;
104 }
