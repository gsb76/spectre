577 template <typename TT, typename... TArgs>
578 class is_callable {
579  // The reason we have private before public here is that we have static member
580  // functions and since this is meant to be a super lightweight helper class
581  // it's better to break convention than increase code size.
582  private:
584  // We pass an int here to disambiguate the two possible templates and have the
585  // compiler prefer the first one. If it cannot be used because there's no
586  // call operator, then it uses the second one.
587  template <typename T, typename... Args>
588  static auto test_callable(int)
589  -> decltype(std::declval<T>()(std::declval<Args>()...), std::true_type());
590 
591  template <typename, typename...>
592  static auto test_callable(...) -> std::false_type;
594 
595  public:
597  static constexpr bool value = decltype(test_callable<TT, TArgs...>(0))::value;
599  using type = std::integral_constant<bool, value>;
600 };
